47inline bool isStandardErrorCode(MBErrorCode code) {
48 switch (code) {
49 case IllegalFunction:
50 case IllegalDataAddress:

54 case SlaveDeviceBusy:

56 case MemoryParityError:

60 case ErrorCodeCRCError:
61 case InvalidCRC:
62 case InvalidByteOrder:
63 case InvalidMessageID:
64 case ProtocolError:
